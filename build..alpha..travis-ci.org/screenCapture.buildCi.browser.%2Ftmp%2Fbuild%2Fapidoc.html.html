    function restore() {
  for (var ext in oldHandlers) {
    if (oldHandlers[ext] === undefined) {
      delete require.extensions[ext];
    } else {
      require.extensions[ext] = oldHandlers[ext];
    }
  }

  exports.oldHandlers = oldHandlers = {};
}
